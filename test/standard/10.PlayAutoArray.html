Description:
Play array of sounds automatically.
Result:
Play two sounds at the same time when loaded.
Code:
            function onload() {
                var engine = new jWebAudio.SoundEngine();
                var sources = engine.addSoundSource({
                    'url': ['../../examples/resource/in.wav',
                            '../../examples/resource/a.ogg'],
                    'preLoad': true,
                    'callback': function() {
                        for (var i in sources) {
                            sources[i].sound.play();
                        }
                    }
                });
            }
