Description:
Play sound automatically.
Result:
Play sound when loaded.
Code:
            function onload() {
                var engine = new jWebAudio.SoundEngine();
                var source = engine.addSoundSource({
                    'url': '../../examples/resource/a.ogg',
                    'preLoad': true,
                    'callback': function() {
                        source.sound.play();
                    }
                });
            }
