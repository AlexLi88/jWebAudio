Description:
Set velocity of 3D Effect.
Result:
Doppler effect change with value of x, y and z input.
Code:
            var effect = null;
            var v = {
                x: 0,
                y: 0,
                z: 0
            };
            
            function onload() {
                $('#audio').jWebAudio('addSoundSource', {
                    'url': '../../examples/resource/a.ogg',
                    'preLoad': true,
                    'callback': function() {
                        $('#audio').jWebAudio('play');
                        var id = $('#audio').jWebAudio('addEffect', '3d');
                        effect = $('#audio').jWebAudio('getEffect', id);
                        
                        // enable UI
                        $('input').each(function() {
                            $(this)[0].disabled = false;
                        });
                    }
                });
            }
            
            function xChange() {
                var x = $('#xInput')[0].value;
                $('xLabel').text(x);
                v.x = parseInt(x);
                effect.soundObject.setVelocity(v.x, v.y, v.z);
            }
            
            function yChange() {
                var y = $('#yInput')[0].value;
                $('xLabel').text(y);
                v.y = parseInt(y);
                effect.soundObject.setVelocity(v.x, v.y, v.z);
            }
            
            function zChange() {
                var z = $('#zInput')[0].value;
                $('zLabel').text(z);
                v.z = parseInt(z);
                effect.soundObject.setVelocity(v.x, v.y, v.z);
            }
