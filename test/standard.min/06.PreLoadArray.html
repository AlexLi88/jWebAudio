Description:
PreLoad two sound sources when add sound source.
Result:
Alert 'Loaded both!'.
Code:
            function onload() {
                var engine = new jWebAudio.SoundEngine();
                var source = engine.addSoundSource({
                    'url': ['../../examples/resource/in.wav',
                            '../../examples/resource/out.wav'],
                    'preLoad': true,
                    'callback': function() {
                        alert('Loaded both!');
                    }
                });
            }
