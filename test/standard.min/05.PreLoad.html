Description:
PreLoad the sound when add sound source.
Result:
Alert 'Loaded!'.
Code:
            function onload() {
                var engine = new jWebAudio.SoundEngine();
                var source = engine.addSoundSource({
                    'url': '../../examples/resource/a.ogg',
                    'preLoad': true,
                    'callback': function() {
                        alert('Loaded!');
                    }
                });
            }
