var source = null;
            function onload() {
                var engine = new jWebAudio.SoundEngine();
                source = engine.addSoundSource({
                    'url': 'resource/a.ogg'
                });
                source.sound.load(function() {
                    source.play();
                });
            }
            
            var useEffect = false;
            var id = null;
            function toggleEffect() {
                useEffect = !useEffect;
                if (useEffect) {
                    id = source.sound.addEffect('telephonize');
                } else if (id !== null) {
                    source.sound.deleteEffect(id);
                }
            }
