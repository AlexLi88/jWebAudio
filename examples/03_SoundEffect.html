$(document).ready(function() {
                $('#div1').jWebAudio('addSoundSource', {
                    'url': 'resource/a.ogg',
                    'preLoad': true,
                    'callback': function() {
                        $('#div1').jWebAudio('play');
                    }
                });
                
                var useEffect = false;
                var id = null;
                $('#effectBtn').click(function() {
                    useEffect = !useEffect;
                    if (useEffect) {
                        id = $('#div1').jWebAudio('addEffect', 'telephonize');
                    } else if (id !== null) {
                        $('#div1').jWebAudio('deleteEffect', id);
                    }
                });
            });
