Description:
Play sound.
Result:
Play sound when click the Play button.
Code:
            function onload() {
                $('#audio').jWebAudio('addSoundSource', {
                    'url': '../../examples/resource/a.ogg'
                });
            }
            
            function play() {
                $('#audio').jWebAudio('load', function() {
                    $('#audio').jWebAudio('play');
                });
                $('#btn')[0].disabled = true;
            }
