Description:
Play array of sounds automatically.
Result:
Play two sounds at the same time when loaded.
Code:
            function onload() {
                var url = {
                    'in': 'in.wav',
                    'a': 'a.ogg'
                };
                $('.sound').each(function() {
                    var id = $(this).attr('id');
                    $(this).jWebAudio('addSoundSource', {
                        url: '../../examples/resource/' + url[id],
                        preLoad: true,
                        callback: function() {
                            $('#' + id).jWebAudio('play');
                        }
                    });
                });
            }
