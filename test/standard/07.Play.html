Description:
Play sound.
Result:
Play sound when click the Play button.
Code:
            var source = null;
            
            function onload() {
                var engine = new jWebAudio.SoundEngine();
                source = engine.addSoundSource({
                    'url': '../../examples/resource/a.ogg'
                });
            }
            
            function play() {
                source.load(function() {
                    source.sound.play();
                });
                document.getElementById('btn').disabled = true;
            }
